import { Directive, HostBinding, HostListener } from '@angular/core';

@Directive({
  selector: '[appDropdown]',
})
export class DropdownDirective {
  @HostBinding('class.open') isOpen = false;

  constructor() {}

  @HostListener('click') toogleOpen() {
    this.isOpen = !this.isOpen;
  }
}

## [Closing the Dropdown From Anywhere](https://www.udemy.com/course/the-complete-guide-to-angular-2/learn/lecture/14906174#overview)

```typescript {3,14,16-20} title="src/app/shared/directives/dropdown.directive.ts"
import {
  Directive,
  ElementRef,
  HostBinding,
  HostListener,
} from '@angular/core';

@Directive({
  selector: '[appDropdown]',
})
export class DropdownDirective {
  @HostBinding('class.open') isOpen = false;

  constructor(private elRef: ElementRef) {}

  @HostListener('document:click', ['$event']) toggleOpen(event: Event) {
    this.isOpen = this.elRef.nativeElement.contains(event.target)
      ? !this.isOpen
      : false;
  }
}
```
