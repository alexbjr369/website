## [Creating a Logging Service](https://www.udemy.com/course/the-complete-guide-to-angular-2/learn/lecture/6656208#overview)

```bash
# npm
npx ng generate service logging/logging

# yarn
yarn ng generate service logging/logging
```

```typescript {9-11} title="src/app/logging/logging.service.ts"
import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root',
})
export class LoggingService {
  constructor() {}

  logStatusChange(status: string) {
    console.log('A server status changed, new status: ' + status);
  }
}
```
