## [Constructor Functions and the new Operator](https://www.udemy.com/course/the-complete-javascript-course/learn/lecture/22649039#overview)

```js title="srcipt.js"
'use strict';

// 1. New {} is created
// 2. function is called, 'this' = {}
// 3. {} linked to prototype
// 4. function automatically returns {}

const Person = function (firstName, birthYear) {
  console.log(this); // result: Person {}
};

new Person('John', 1986);
```

```js title="srcipt.js"
'use strict';

const Person = function (firstName, birthYear) {
  // instance properties
  this.firstName = firstName;
  this.birthYear = birthYear;
};

const person = new Person('John', 1986);

console.log(person); // result: Person {firstName: 'John', birthYear: 1986}
console.log(person instanceof Person); // result: true
```

```js title="srcipt.js"
'use strict';

const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;

  // bad practice: never create a method inside a constructor function
  // it will create a copy of this method and attach it into every single object
  this.calcAge = function () {
    console.log(2036 - this.birthYear);
  };
};

const person = new Person('John', 1986);

console.log(person);
```

## [Prototypes](https://www.udemy.com/course/the-complete-javascript-course/learn/lecture/22649045#overview)

```js title="srcipt.js"
'use strict';

const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;

  // this.calcAge = function () {
  // console.log(2036 - this.birthYear);
  // };
};

// there will be only one copy of this method and all the objects created from Person
// will have access to it
Person.prototype.calcAge = function () {
  console.log(2036 - this.birthYear);
};

const person = new Person('John', 1986);

console.log(person.calcAge()); // result: 50
```

```js title="srcipt.js"
'use strict';

// 3. {} linked to prototype, it will create the '__proto__' property and it sets its value to the
// prototype property of the function that is being called
// person.__proto__ === Person.prototype

const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;
};

Person.prototype.calcAge = function () {
  console.log(2036 - this.birthYear);
};

const person = new Person('John', 1986);

console.log(person.__proto__); // result: {calcAge: ƒ, constructor: ƒ}
console.log(person.__proto__ === Person.prototype); // result: true

// Person.prototype is not the prototype of Person, but it will be used as the prototype of all
// objects created with the Person constructor function
console.log(Person.prototype.isPrototypeOf(person)); // result: true
console.log(Person.prototype.isPrototypeOf(Person)); // result: false
```

```js title="srcipt.js"
'use strict';

const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;
};

const person = new Person('John', 1986);

Person.prototype.species = 'Homo Sapiens';
console.log(person.species); // result: 'Homo Sapiens';

console.log(person.hasOwnproperty('species')); // result: false, the property is in 'Person' prototype
console.log(person.hasOwnproperty('firstName')); // result: true
```

## [Prototypal Inheritance on Built-In Objects](https://www.udemy.com/course/the-complete-javascript-course/learn/lecture/22649059#overview)

```js title="srcipt.js"
'use strict';

const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;
};

const person = new Person('John', 1986);

console.log(person.__proto__.__proto__); // result: Object
console.log(person.__proto__.__proto__.__proto__); // result: null
console.dir(Person.prototype.constructor); // result: Person(firstName, birthYear)
```

```js title="srcipt.js"
// extending built-in objects (never do it)
const array = [1, 2, 3, 3, 2];
Array.prototype.unique = function () {
  return [...new Set(this)];
};

console.log(array.unique()); // result: [1, 2, 3]
```

## [Setters and Getters](https://www.udemy.com/course/the-complete-javascript-course/learn/lecture/22649073#overview)

```js title="srcipt.js"
// setter and getters are called accessor functions

const account = {
  owner: 'John',
  movements: [100, 300, 400],
  get latest() {
    return this.movements.slice(-1).pop();
  },
  set latest(movement) {
    this.movements.push(movement);
  },
};

// getter
console.log(account.latest); // result: 400
// setter
account.latest = 50;
console.log(account.movements); // result: [100, 300, 400, 50]
```

```js title="srcipt.js"
class Person {
  constructor(firstName, birthYear) {
    this.firstName = firstName;
    this.birthYear = birthYear;
  }

  // methods are added to the 'prototype' property
  calcAge() {
    console.log(2036 - this.birthYear);
  }

  get age() {
    return 2036 - this.birthYear;
  }
}

const person = new Person('John', 1986);
console.log(person.age); // result: 50
```

```js title="srcipt.js"
class Person {
  constructor(fullName, birthYear) {
    this.fullName = fullName;
    this.birthYear = birthYear;
  }

  // methods are added to the 'prototype' property
  calcAge() {
    console.log(2036 - this.birthYear);
  }

  get age() {
    return 2036 - this.birthYear;
  }

  set fullName(name) {
    // '_fullName' convention in case we are setting a property that already exists
    if (name.includes(' ')) this._fullName = name;
    else alert(`${name} is not a full name!`);
  }

  // creation of a getter of '_fullName'
  get fullName() {
    return this._fullName;
  }
}

const person = new Person('John Doe', 1986);
console.log(person);
```

## [Static Methods](https://www.udemy.com/course/the-complete-javascript-course/learn/lecture/22649077#overview)

**Static Methods are attached to the constructor function and not the prototype.**

```js title="srcipt.js"
Array.from();

Number.parseFloat();
```

```js title="srcipt.js"
// constructor function
const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;
};

Person.hey = function () {
  console.log('hey there!');
};

Person.hey();

const person = new Person('John', 1986);
// not inherited
person.hey();
```

```js title="srcipt.js"
// classes
class Person {
  constructor(fullName, birthYear) {
    this.fullName = fullName;
    this.birthYear = birthYear;
  }

  static hey() {
    console.log('hey there!');
  }
}

Person.hey();

const person = new Person('John', 1986);
// not inherited
person.hey();
```

## [Object.create](https://www.udemy.com/course/the-complete-javascript-course/learn/lecture/22649081#overview)

```js title="srcipt.js"
const PersonPrototype = {
  calcAge() {
    console.log(2036 - this.birthYear);
  },
};

const person = Object.create(PersonPrototype);
// not good
person.firstName = 'John';
person.birthYear = 1986;

person.calcAge(); // result: 50
```

```js title="srcipt.js"
const PersonPrototype = {
  calcAge() {
    console.log(2036 - this.birthYear);
  },
};

const person = Object.create(PersonPrototype);

console.log(person.__proto__); // result: {calcAge:}
console.log(person.__proto__ === PersonPrototype); // result: true
```

```js title="srcipt.js"
const PersonPrototype = {
  calcAge() {
    console.log(2036 - this.birthYear);
  },
  // good way of creating object properties
  init(firstName, birthYear) {
    this.firstName = firstName;
    this.birthYear = birthYear;
  },
};

const person = Object.create(PersonPrototype);
person.init('John', 1986);

person.calcAge(); // result: 50
```
